# Monorepo push akışı (dev -> PR -> main)  [PowerShell]

# 0) Repo root'ta olduğundan emin ol (VampireVape klasörü)
cd C:\Users\murat\Desktop\MC\MC\EC\VampireVape

# 1) Değişiklikleri doğrula
git status
git diff

# 2) dev branch'e geç (yoksa oluştur)
git checkout dev 2>$null
if ($LASTEXITCODE -ne 0) { git checkout -b dev }

# 3) Her şeyi stage et (sadece apps görünüyor sanıyorsan muhtemelen diğer dosyalar değişmemiştir)
git add -A

# 4) Commit
git commit -m "feat: headless storefront + shopify integration" 2>$null

# 5) Remote ayarlı mı kontrol et (yoksa ekle)
git remote -v
# git remote add origin https://github.com/murathan-cotuk/vampirevape.git

# 6) dev branch'i pushla
git push -u origin dev

# 7) GitHub'da Pull Request aç: dev -> main
# 8) PR merge (Squash veya Merge), sonra local main'i güncelle
git checkout main
git pull origin main
